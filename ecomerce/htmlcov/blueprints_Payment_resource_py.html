148 parser.add_argument('sort', location = 'args', help = 'invalid sort value', choices = ('desc','asc'))

152 offset = (args['p'] * args['rp']) - args['rp'] 

153 

154 qry = Payments.query 

155 

156 if args['checkout_id'] is not None: 

157 qry = qry.filter_by(checkout_id = args['checkout_id']) 

158 

159 if args['orderby'] is not None : 

160 if args['orderby'] == 'checkout_id': 

161 if args['sort'] == 'desc': 

162 qry = qry.order_by(desc(Payments.checkout_id)) 

163 else: 

164 qry = qry.order_by(Payments.checkout_id) 

165 rows = [] 

166 for row in qry.limit(args['rp']).offset(offset).all(): 

167 if row is not None and row.deleted == False:
